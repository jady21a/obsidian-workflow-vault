```dataviewjs
const currentPath = dv.current().file.path;
const targetTags = new Set(["book", "task", "project", "todo"]);
const tagOrder = {book: 1, project: 2, task: 3, todo: 4};
const cutoff = dv.date("today").minus({days: 7});

const groups = new Map();

for (const page of dv.pages("")) {
  if (page.file.path === currentPath) continue;
  if (page.file.mtime < cutoff) continue;
  
  for (const rawTag of (page.file.tags || [])) {
    const tag = rawTag.startsWith("#") ? rawTag.slice(1) : rawTag;
    if (!targetTags.has(tag) || tag.length > 10) continue;
    if (!groups.has(tag)) groups.set(tag, []);
    groups.get(tag).push(page);
  }
}

const rows = [...groups.entries()]
  .sort(([a, ap], [b, bp]) => {
    const diff = (tagOrder[a] ?? 5) - (tagOrder[b] ?? 5);
    if (diff !== 0) return diff;
    return Math.max(...bp.map(p => p.file.size)) - Math.max(...ap.map(p => p.file.size));
  })
  .map(([tag, pages]) => {
    pages.sort((a, b) => b.file.size - a.file.size);
    return [
      tag,
      pages.map(p => p.file.link),
      pages.map(p => Math.round(p.file.size / 1024)),
      pages.map(p => p.file.mtime.toFormat("MM-dd HH:mm")),
      pages.length
    ];
  });

dv.table(["tag", "笔记", "大小(KB)", "修改时间", "数量"], rows);
```






可以显示标签+按文件大小排列
```dataviewjs
function normalizeTags(p) {
    if (!p.tags) return [];                  // 无标签
    return Array.isArray(p.tags) ? p.tags : [p.tags]; // 字符串转数组
}

const categories = ["book", "project", "task", "todo"];
const sevenDaysAgo = dv.date("today").minus({days: 7});
let allRows = [];

for (let cat of categories) {
    let pages = dv.pages("")
        .where(p => p.file.mtime >= sevenDaysAgo && normalizeTags(p).some(t => t.includes(cat)))
        .sort(p => p.file.size, "desc");
    
    if (pages.length > 0) {
        allRows.push([cat, "", "", "", "", pages.length]);
        pages.forEach(p => {
            allRows.push(["", p.file.link, p.file.size, dv.date(p.file.mtime).toFormat("MM-dd HH:mm"), "", ""]);
        });
    }
}

// 显示有其他标签的笔记
let otherTagPages = dv.pages("")
    .where(p => {
        if (p.file.mtime < sevenDaysAgo) return false;
        let tags = normalizeTags(p);
        if (tags.length === 0) return false;
        let hasMainTag = tags.some(t => categories.some(cat => t.includes(cat)));
        return !hasMainTag;
    })
    .sort(p => p.file.size, "desc");

if (otherTagPages.length > 0) {
    allRows.push(["其他标签", "", "", "", "", otherTagPages.length]);
    otherTagPages.forEach(p => {
        allRows.push(["", p.file.link, p.file.size, dv.date(p.file.mtime).toFormat("MM-dd HH:mm"), normalizeTags(p).join(", "), ""]);
    });
}

// 显示没有标签的笔记
let noTagPages = dv.pages("")
    .where(p => p.file.mtime >= sevenDaysAgo && normalizeTags(p).length === 0)
    .sort(p => p.file.size, "desc");

if (noTagPages.length > 0) {
    allRows.push(["无标签", "", "", "", "", noTagPages.length]);
    noTagPages.forEach(p => {
        allRows.push(["", p.file.link, p.file.size, dv.date(p.file.mtime).toFormat("MM-dd HH:mm"), "", ""]);
    });
}

dv.table(["tag", "笔记", "大小(KB)", "修改时间", "标签", "数量"], allRows);
```
