# 项目总览仪表盘


## 所有项目
```dataview
TABLE
  status as "状态",
  priority as "优先级",
  start-date as "开始日期",
  end-date as "预计完成",
  description as "描述"
FROM "4.Projects"
WHERE type = "main-project"
SORT priority ASC, file.name ASC
```

## 高优先级项目
```dataview
TABLE WITHOUT ID
  file.link as "项目",
  status as "状态",
  end-date as "预计完成",
  description as "描述"
FROM "4.Projects"
WHERE type = "main-project" AND priority = "high" AND status != "已完成"
SORT file.name ASC
```

## 任务动态
```dataviewjs
// ===== 公共 helper(数据源统一为 try[],空时回退页面字段)=====
const MOOD = {
  a: { score: 4, emoji: "😊", label: "非常好" },
  b: { score: 3, emoji: "🙂", label: "良好" },
  c: { score: 2, emoji: "😐", label: "一般" },
  d: { score: 1, emoji: "😔", label: "不佳" },
};
const moodKey = (m) => { m = Array.isArray(m) ? m[0] : m; return m ? String(m).toLowerCase() : null; };
const bar = (n) => { n = Math.max(0, Math.min(10, Math.round(n || 0))); return "█".repeat(n) + "░".repeat(10 - n); };
const ms = (d) => { // 转毫秒,兼容 DateTime / Date / number / 字符串
  if (d == null) return null;
  if (typeof d === "number") return d;
  if (typeof d.toMillis === "function") return d.toMillis();
  if (d instanceof Date) return d.getTime();
  const x = dv.date(String(d));
  return x ? x.toMillis() : null;
};

function records(page) {
  const valid = (page.try || []).filter((b) => b != null);
  if (valid.length) {
    return valid.map((b) => ({
      page, project: page.project, sub: page["sub-project"],
      date: b.日期, completion: b.完成度, mood: b.心情,
    }));
  }
  return [{ page, project: page.project, sub: page["sub-project"],
            date: page.date, completion: page.completion, mood: page.mood }];
}
const latest = (page) => records(page)
  .sort((a, b) => (ms(a.date) || 0) - (ms(b.date) || 0)).at(-1);

const allPages = dv.pages('"4.Projects/II.tasks"');
const nowMs = dv.date("now").toMillis();

// ---------- 📈 本周进度(最近 7 天的 try 记录)----------
dv.header(3, "📈 本周进度");
const weekAgoMs = nowMs - 7 * 864e5;
const recent = allPages.flatMap(records)
  .filter((r) => ms(r.date) != null && ms(r.date) >= weekAgoMs)
  .sort((r) => ms(r.date), "desc");

if (recent.length === 0) {
  dv.paragraph("_最近 7 天暂无任务记录_");
} else {
  dv.table(["日期", "项目", "子项目", "完成度", "心情"],
    recent.map((r) => {
      const k = moodKey(r.mood);
      return [r.date, r.project, r.sub,
        `\`${bar(r.completion)}\` ${r.completion ?? "—"}/10`,
        k && MOOD[k] ? `${MOOD[k].emoji} ${k.toUpperCase()}` : ""];
    }));
  const arr = recent.array();
  const avg = arr.reduce((s, r) => s + (r.completion || 0), 0) / arr.length;
  const ks = arr.map((r) => moodKey(r.mood)).filter((k) => k && MOOD[k]);
  const moodText = ks.length === 0 ? "—" : (() => {
    const a = ks.reduce((s, k) => s + MOOD[k].score, 0) / ks.length;
    return a > 3.5 ? "😊 良好" : a > 2.5 ? "😐 一般" : "😔 需要调整";
  })();
  dv.paragraph([`**本周记录数**: ${recent.length}`,
    `**平均完成度**: ${avg.toFixed(1)}/10`, `**平均心情**: ${moodText}`].join("  \n"));
}

// ---------- ⚠️ 需要关注的任务(最近一次完成度 < 5)----------
dv.header(3, "⚠️ 需要关注的任务 (完成度 < 5)");
const twoWeeksMs = nowMs - 14 * 864e5;
const flagged = allPages.map(latest)
  .filter((r) => r && ms(r.date) != null && ms(r.date) >= twoWeeksMs && (r.completion ?? 99) < 5)
  .sort((r) => ms(r.date), "desc")
  .slice(0, 10);

if (flagged.length === 0) {
  dv.paragraph("_近 14 天无低完成度任务_");
} else {
  dv.table(["任务", "项目", "完成度", "日期", "心情"],
    flagged.map((r) => {
      const k = moodKey(r.mood);
      return [r.page.file.link, r.project, r.completion, r.date,
        k && MOOD[k] ? `${MOOD[k].emoji} ${k.toUpperCase()}` : ""];
    }));
}

// ---------- 😊 心情趋势(最近 30 天)----------
dv.header(3, "😊 心情趋势");
const monthAgoMs = nowMs - 30 * 864e5;
const counts = { a: 0, b: 0, c: 0, d: 0 };
allPages.flatMap(records)
  .filter((r) => ms(r.date) != null && ms(r.date) >= monthAgoMs)
  .forEach((r) => { const k = moodKey(r.mood); if (k && counts[k] !== undefined) counts[k]++; });

const total = counts.a + counts.b + counts.c + counts.d;
if (total === 0) {
  dv.paragraph("_最近 30 天暂无心情记录_");
} else {
  const lines = ["a", "b", "c", "d"].map((k) => {
    const n = counts[k];
    const blocks = Math.round((n / total) * 20);
    const pct = ((n / total) * 100).toFixed(0);
    return `${MOOD[k].emoji} ${MOOD[k].label} (${k.toUpperCase()})  \`${"█".repeat(blocks).padEnd(20, "░")}\`  ${n} 次 · ${pct}%`;
  });
  dv.paragraph(`**最近30天心情分布**(共 ${total} 条)\n\n${lines.join("\n\n")}`);
}
```

---
## 数据说明
- **完成度 (1-10)**:1-3 刚开始 · 4-6 进行中 · 7-8 接近完成 · 9-10 基本完成
- **心情 (a-d)**:A 非常好 · B 良好 · C 一般 · D 不佳
- **数据源**:统计与动态均取每个任务 `try[]` 数组(每日记录);`try` 为空时回退页面级字段。
